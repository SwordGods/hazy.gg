-- Função para verificar a whitelist
local function checkWhitelist()
    local whitelistUrl = "https://pastebin.com/XWxATggK"
    local whitelist = HttpGet(whitelistUrl)

    -- Verifica se o nome do jogador está na whitelist
    local playerName = game.Players.LocalPlayer.Name
    if string.find(whitelist, playerName) then
        return true  -- Jogador está na whitelist
    else
        return false  -- Jogador não está na whitelist
    end
end

-- Função principal que executa o script
local function executeScript()
    local isWhitelisted = checkWhitelist()

    if isWhitelisted then
        -- Jogador está na whitelist, executa o código e dá o item
        local item = game.ReplicatedStorage.Game_Item["UZI"]
        item:Clone().Parent = game.Players.LocalPlayer.Backpack
        
        -- Envia a notificação para o jogador que foi aceito na whitelist
        wait(0.2)  -- Aguarda um pequeno tempo para garantir que a notificação apareça após a execução
        game.StarterGui:SetCore("SendNotification", {
            Title = "Hazy.GG | Mensagem",  -- Título da notificação
            Text = "Whitelist aceita, cliente iniciado.",  -- Texto da notificação
            Duration = 5  -- Duração da notificação em segundos
        })
    else
        -- Jogador não está na whitelist, dá kick com a mensagem
        game.Players.LocalPlayer:Kick("Seu nome não foi encontrado na whitelist do Hazy.GG, se alguém compartilhou o script com você a pessoa irá perder a whitelist.")
    end
end

-- Executa o script
executeScript()
